// index.html 中的 initCoze 函数 async function initCoze(userName) { // 1. 请求带有 session_name 标识的专属 Token const res = await fetch(`/api/getToken?name=${encodeURIComponent(userName)}`); const data = await res.json(); new CozeWebSDK.WebChatClient({ config: { bot_id: '7628621142705831979', isIframe: false // 建议设为 false 以便前端代码能清理本地缓存 }, auth: { type: 'token', token: data.token, onRefreshToken: () => data.token }, // 2. 传入用户信息，确保 UI 显示和记录归属一致 userInfo: { id: userName, nickname: userName }, }).showChatBot(); }
